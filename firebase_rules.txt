rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    
    // Allow collection group queries on decks for authenticated users
    match /{path=**}/decks/{deckId} {
      allow read: if request.auth != null;
    }
    
    // Rules for users and their decks/cards
    match /users/{userId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && request.auth.uid == userId;
      
      match /decks/{deckId} {
        allow read: if request.auth != null;
        allow write: if request.auth != null && request.auth.uid == userId;
        
        match /cards/{cardId} {
          allow read: if request.auth != null;
          allow write: if request.auth != null && request.auth.uid == userId;
        }
      }
    }
  }
}
